        function LonelinessSanctuary({ onClose }) {
            const [textIndex, setTextIndex] = React.useState(0);
            const [particles, setParticles] = React.useState([]);

            const messages = [
                "You are safe here.",
                "This feeling is temporary.",
                "Your worth is not defined by productivity.",
                "You are enough, just as you are.",
                "Breathe in... and let go.",
                "The stars shine brightest in the dark."
            ];

            React.useEffect(() => {
                const interval = setInterval(() => {
                    setTextIndex((prev) => (prev + 1) % messages.length);
                }, 4000);
                return () => clearInterval(interval);
            }, []);

            const handleRelease = (e) => {
                const rect = e.target.getBoundingClientRect();
                const newParticles = [];
                for (let i = 0; i < 30; i++) {
                    const angle = Math.random() * Math.PI * 2;
                    const velocity = 50 + Math.random() * 100;
                    newParticles.push({
                        id: Date.now() + i,
                        tx: Math.cos(angle) * velocity + 'px',
                        ty: Math.sin(angle) * velocity + 'px'
                    });
                }
                setParticles(newParticles);
                setTimeout(() => setParticles([]), 1000);
            };

            return (
                <div className="sanctuary-overlay">
                    {/* Background Stars */}
                    {[...Array(50)].map((_, i) => (
                        <div
                            key={i}
                            className="star"
                            style={{
                                top: Math.random() * 100 + '%',
                                left: Math.random() * 100 + '%',
                                width: Math.random() * 3 + 1 + 'px',
                                height: Math.random() * 3 + 1 + 'px',
                                '--duration': Math.random() * 3 + 2 + 's',
                                '--delay': Math.random() * 5 + 's'
                            }}
                        ></div>
                    ))}

                    <div className="breathing-circle">
                        <LucideIcon name="wind" size={48} color="rgba(255,255,255,0.8)" />
                    </div>

                    <div key={textIndex} className="comfort-text" style={{ animation: 'floatText 4s ease-in-out' }}>
                        {messages[textIndex]}
                    </div>

                    <div style={{ position: 'relative' }}>
                        <button className="release-btn" onClick={handleRelease}>
                            Release Stress
                        </button>
                        {particles.map(p => (
                            <div
                                key={p.id}
                                className="particle"
                                style={{
                                    left: '50%',
                                    top: '50%',
                                    '--tx': p.tx,
                                    '--ty': p.ty,
                                    animation: 'particleDissolve 0.8s ease-out forwards'
                                }}
                            ></div>
                        ))}
                    </div>

                    <button
                        onClick={onClose}
                        style={{
                            position: 'absolute',
                            top: '40px',
                            right: '40px',
                            background: 'transparent',
                            border: 'none',
                            color: 'rgba(255,255,255,0.5)',
                            cursor: 'pointer',
                            fontSize: '14px',
                            display: 'flex',
                            alignItems: 'center',
                            gap: '8px'
                        }}
                    >
                        <LucideIcon name="x" size={20} /> Exit Sanctuary
                    </button>
                </div>
            );
        }

        function ChatInterface({ onNavigate }) {
            const [messages, setMessages] = React.useState([
                {
                    id: 1,
                    text: "Hello! I'm your AI mental health companion. I can detect emotions from how you type, not just what you type. How are you feeling today?",
                    sender: 'bot',
                    timestamp: new Date(),
                    typingMetrics: null
                }
            ]);
            const [inputText, setInputText] = React.useState('');
            const [isTyping, setIsTyping] = React.useState(false);
            const [typingMetrics, setTypingMetrics] = React.useState({
                speed: 0,
                backspaces: 0,
                pauses: 0,
                capsBursts: 0,
                currentMood: 'neutral'
            });
            const [crisisScore, setCrisisScore] = React.useState(25);
            const [showEmergencyModal, setShowEmergencyModal] = React.useState(false);
            const [lonelinessMode, setLonelinessMode] = React.useState(false);

            const messagesEndRef = React.useRef(null);
            const typingTimerRef = React.useRef(null);
            const [lastTypingTime, setLastTypingTime] = React.useState(Date.now());
            const [isListening, setIsListening] = React.useState(false);
            const recognitionRef = React.useRef(null);

            React.useEffect(() => {
                if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
                    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
                    recognitionRef.current = new SpeechRecognition();
                    recognitionRef.current.continuous = true;
                    recognitionRef.current.interimResults = true;

                    recognitionRef.current.onresult = (event) => {
                        let finalTranscript = '';
                        for (let i = event.resultIndex; i < event.results.length; ++i) {
                            if (event.results[i].isFinal) {
                                finalTranscript += event.results[i][0].transcript;
                            }
                        }
                        if (finalTranscript) {
                            setInputText(prev => prev + (prev ? ' ' : '') + finalTranscript);
                            analyzeTypingBehavior(finalTranscript, { key: 'VoiceInput' });
                        }
                    };

                    recognitionRef.current.onerror = (event) => {
                        console.error('Speech recognition error', event.error);
                        setIsListening(false);
                    };

                    recognitionRef.current.onend = () => {
                        setIsListening(false);
                    };
                }
            }, []);

            const toggleListening = () => {
                if (isListening) {
                    recognitionRef.current.stop();
                    setIsListening(false);
                } else {
                    if (recognitionRef.current) {
                        recognitionRef.current.start();
                        setIsListening(true);
                    } else {
                        alert("Voice input is not supported in this browser. Please try Chrome or Edge.");
                    }
                }
            };

            React.useEffect(() => {
                messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
            }, [messages]);

            const analyzeTypingBehavior = (text, event) => {
                const now = Date.now();
                const timeDiff = now - lastTypingTime;

                if (timeDiff > 1000) {
                    setTypingMetrics(prev => ({ ...prev, pauses: prev.pauses + 1 }));
                }

                if (event.key === 'Backspace' || event.key === 'Delete') {
                    setTypingMetrics(prev => ({ ...prev, backspaces: prev.backspaces + 1 }));
                }

                if (event.key.length === 1 && event.key === event.key.toUpperCase() && !event.shiftKey) {
                    if (text.toUpperCase() === text) {
                        setTypingMetrics(prev => ({ ...prev, capsBursts: prev.capsBursts + 1 }));
                    }
                }

                setLastTypingTime(now);

                if (typingTimerRef.current) {
                    clearTimeout(typingTimerRef.current);
                }

                typingTimerRef.current = setTimeout(() => {
                    const speed = text.length > 0 ? text.length / 5 : 0;
                    const finalMetrics = { ...typingMetrics, speed };

                    let mood = 'neutral';
                    if (finalMetrics.backspaces > 5) mood = 'anxious';
                    if (finalMetrics.pauses > 3) mood = 'overwhelmed';
                    if (finalMetrics.capsBursts > 2) mood = 'agitated';
                    if (finalMetrics.speed < 1.0) mood = 'fatigued';
                    if (finalMetrics.speed > 2.0 && finalMetrics.backspaces < 2 && finalMetrics.pauses < 2) mood = 'calm';

                    setTypingMetrics({ ...finalMetrics, currentMood: mood });

                    const moodScores = {
                        calm: -10,
                        neutral: 0,
                        anxious: 15,
                        overwhelmed: 20,
                        agitated: 10,
                        fatigued: 12
                    };

                    const newScore = Math.max(0, Math.min(100, crisisScore + (moodScores[mood] || 0)));
                    setCrisisScore(newScore);

                    // Check for loneliness triggers
                    const hour = new Date().getHours();
                    const isLateNight = hour >= 22 || hour <= 4;
                    if ((isLateNight && text.toLowerCase().includes('alone')) ||
                        (text.length < 10 && text.toLowerCase().includes('tired'))) {
                        setLonelinessMode(true);
                    }

                    // Show emergency modal if score is high
                    if (newScore > 70) {
                        setTimeout(() => setShowEmergencyModal(true), 1000);
                    }
                }, 2000);
            };

            const handleInputChange = (e) => {
                setInputText(e.target.value);
                analyzeTypingBehavior(e.target.value, e);
            };

            const [newWord, setNewWord] = React.useState('');
            const [conflictText, setConflictText] = React.useState('');
            const [rewrittenText, setRewrittenText] = React.useState('');

            const handleAddWord = () => {
                if (newWord.trim() && !blockedWords.includes(newWord.trim().toLowerCase())) {
                    setBlockedWords([...blockedWords, newWord.trim().toLowerCase()]);
                    setNewWord('');
                }
            };

            const handleRemoveWord = (word) => {
                setBlockedWords(blockedWords.filter(w => w !== word));
            };

            const handleRewrite = () => {
                if (!conflictText.trim()) return;

                const rewritten = conflictText
                    .replace(/you never/g, "I feel unheard when")
                    .replace(/you always/g, "I notice that often")
                    .replace(/you're wrong/g, "I see it differently")
                    .replace(/you don't understand/g, "I'd like to help you understand my perspective")
                    .replace(/you're/g, "I feel when")
                    .replace(/you/g, "I");

                setRewrittenText(rewritten);
            };

            return (
                <div className="settings-page">
                    <nav className="dashboard-nav">

        // ==================== MAIN APP ====================
        function App() {
            const [currentPage, setCurrentPage] = React.useState('loading');
            const [loadingComplete, setLoadingComplete] = React.useState(false);
            const [navParams, setNavParams] = React.useState({});

            React.useEffect(() => {
                // Simulate loading
                const timer = setTimeout(() => {
                    setLoadingComplete(true);
                    checkOnboarding();
                }, 3000);

                return () => clearTimeout(timer);
            }, []);

            const checkOnboarding = () => {
                const onboardingComplete = localStorage.getItem('onboardingComplete');
                if (onboardingComplete === 'true') {
                    setCurrentPage('dashboard');
                } else {
                    setCurrentPage('onboarding');
                }
            };

            const handleNavigate = (page, params = {}) => {
                setCurrentPage(page);
                setNavParams(params);
            };

            const handleOnboardingComplete = () => {
                setCurrentPage('dashboard');
            };

            if (!loadingComplete) {
                return <LoadingScreen />;
            }

            switch (currentPage) {
                case 'onboarding':
                    return <Onboarding onComplete={handleOnboardingComplete} />;
                case 'dashboard':
                    return <Dashboard onNavigate={handleNavigate} />;
                case 'chat':
                    return <ChatInterface onNavigate={handleNavigate} initialMode={navParams.activeMode} />;
                case 'analytics':
                    return <AnalyticsDashboard onNavigate={handleNavigate} />;
                case 'settings':
                    return <SettingsPage onNavigate={handleNavigate} />;
                default:
                    return <Dashboard onNavigate={handleNavigate} />;
            }
        }

        // ==================== RENDER APP ====================
        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
